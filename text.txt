/* Запуск слайдера "Схожі статті" на сторінці блогу */
const similarBlogSlider = document.querySelector('.similar-blog-slider');
if (similarBlogSlider) {
    new Swiper('.similar-blog-slider', {
        slidesPerView: 1.1, /* На мобілці показуємо 1 цілу статтю і шматочок наступної */
        spaceBetween: 20,
        pagination: {
            el: '.swiper-pagination',
            clickable: true,
        },
        breakpoints: {
            1024: {
                slidesPerView: 3, /* На десктопі показуємо 3 статті */
                spaceBetween: 30,
            }
        }
    });
}
