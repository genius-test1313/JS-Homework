// === ПУНКТ 4: Скрипт для відкриття/закриття мобільних фільтрів ===
document.addEventListener('DOMContentLoaded', function() {
    const filterBtn = document.querySelector('.brands__filter-btn-mobile');
    const filterPanel = document.querySelector('.filters');
    // Беремо заголовок панелі як кнопку для закриття
    const closeBtn = document.querySelector('.filters__heading-title'); 

    if (filterBtn && filterPanel) {
        // Відкрити фільтри по кліку на кнопку
        filterBtn.addEventListener('click', function() {
            filterPanel.classList.add('active');
            document.body.style.overflow = 'hidden'; // Забороняємо скрол фону
        });
    }

    if (closeBtn && filterPanel) {
        // Закрити фільтри по кліку на заголовок з хрестиком
        closeBtn.addEventListener('click', function() {
            filterPanel.classList.remove('active');
            document.body.style.overflow = ''; // Повертаємо скрол фону
        });
    }
});
