/* === ПУНКТ 4: Оживляем мобильные фильтры === */
@media screen and (max-width: 991px) {
    .filters {
        position: fixed;
        top: 0;
        left: -100%; /* Полностью прячем за левый край */
        width: 100%;
        height: 100vh;
        background: #fff;
        z-index: 1000;
        transition: all 0.4s ease; /* Плавное появление */
        overflow-y: auto;
        padding: 20px;
    }

    /* Когда добавляем этот класс через JS, фильтры выезжают */
    .filters.active {
        left: 0;
    }
}












// Код для открытия фильтров
const filterBtn = document.querySelector('.brands__filter-btn-mobile');
const filterPanel = document.querySelector('.filters');

if (filterBtn && filterPanel) {
    filterBtn.addEventListener('click', function() {
        filterPanel.classList.toggle('active');
        // Запрещаем скролл сайта, когда открыты фильтры
        document.body.classList.toggle('no-scroll');
    });
}
